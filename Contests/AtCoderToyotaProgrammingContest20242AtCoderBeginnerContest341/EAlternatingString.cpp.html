18 int main() {
19   cin.tie(nullptr)->sync_with_stdio(false);
20   cout << fixed << setprecision(10);
21 
22   int n, q;
23   cin >> n >> q;
24   string s;
25   cin >> s;
26   set<int> S;
27   for (int i = 0; i < n - 1; i++) {
28     if (s[i] == s[i + 1]) {
29       S.insert(i);
30     }
31   }
32   while (q--) {
33     int t, l, r;
34     cin >> t >> l >> r;
35     --l;
36     --r;
37     if (t == 1) {
38       if (l - 1 >= 0) {
39         if (S.count(l - 1) > 0) {
40           S.erase(S.find(l - 1));
41         } else {
42           S.insert(l - 1);
43         }
44       }
45       if (S.count(r) > 0) {
46         S.erase(S.find(r));
47       } else {
48         S.insert(r);
49       }
50     } else {
51       auto it = S.lower_bound(l);
52       if (it == S.end() || *it >= r) {
53         cout << "Yes\n";
54       } else {
55         cout << "No\n";
56       }
57     }
58   }
59   return 0;
60 }
